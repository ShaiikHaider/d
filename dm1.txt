1a- load data and identify independent and dependent 
import pandas as pd

# Create a small dataset manually
data = {
    'Age': [22, 25, 28, 30, 35],
    'Salary': [25000, 30000, 35000, 40000, 50000],
    'Experience': [1, 3, 4, 6, 8],
    'Purchased': ['No', 'No', 'Yes', 'Yes', 'Yes']   # Target variable
}

# Create DataFrame
df = pd.DataFrame(data)

# Display dataset
print("Dataset:\n", df)

# Identify independent & dependent variables
independent_vars = df.columns[:-1]   # all except last column
dependent_var = df.columns[-1]       # last column

print("\nIndependent Variables:", list(independent_vars))
print("Dependent Variable:", dependent_var)








1b- handle missing data using mean imputation
import pandas as pd

# Create sample DataFrame with missing values
df = pd.DataFrame({
    'A': [1, 2, None, 4],
    'B': [5, None, 7, None]
})

print("Original DataFrame:\n", df)

# Fill missing values with mean of each column
df = df.fillna(df.mean())

print("\nAfter Mean Imputation:\n", df)

1d- cosine
import numpy as np
from numpy.linalg import norm

x = np.array([1, 2, 3])
y = np.array([4, 5, 6])

cos_sim = np.dot(x, y) / (norm(x) * norm(y))
print(f"Cosine Similarity: {cos_sim}")
