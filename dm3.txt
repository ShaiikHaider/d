3(a) Apply K-Means Clustering and Plot Clusters

import numpy as np
import matplotlib.pyplot as plt
from sklearn.cluster import KMeans

# Create simple dataset (2D points)
X = np.array([
    [1, 1], [1.5, 2], [3, 4],
    [5, 7], [3.5, 5], [4.5, 5],
    [3.5, 4.5]
])

print("Dataset:\n", X)

# Apply K-Means clustering
kmeans = KMeans(n_clusters=2, random_state=0)
kmeans.fit(X)

print("\nCluster Centers:\n", kmeans.cluster_centers_)
print("\nCluster Labels:\n", kmeans.labels_)

# Plot the clusters
plt.scatter(X[:, 0], X[:, 1], c=kmeans.labels_, cmap='rainbow')
plt.scatter(kmeans.cluster_centers_[:, 0], kmeans.cluster_centers_[:, 1],
            color='black', marker='X', s=200, label='Centroids')
plt.title("K-Means Clustering")
plt.xlabel("X1")
plt.ylabel("X2")
plt.legend()
plt.show()




3(b) Implement Apriori Algorithm â€“ Find Frequent Itemsets & Rules


from apyori import apriori

# Sample transactional data
transactions = [
    ['milk', 'bread', 'eggs'],
    ['milk', 'bread'],
    ['milk', 'eggs'],
    ['bread', 'butter'],
    ['milk', 'bread', 'butter', 'eggs']
]

# Apply Apriori algorithm
rules = apriori(transactions,
                min_support=0.3,
                min_confidence=0.6,
                min_lift=1.2,
                min_length=2)

results = list(rules)

# Display results
for item in results:
    pair = item.items
    items = [x for x in pair]
    print("Rule: " + items[0] + " -> " + items[1] if len(items) > 1 else "")
    print("Support:", round(item.support, 3))
    print("Confidence:", round(item.ordered_statistics[0].confidence, 3))
    print("Lift:", round(item.ordered_statistics[0].lift, 3))
    print("=====================================")
